class Solution:
	def longestPalindrome(self, s:str)->str:
		res1 = s[0]
		res2 = ''
		check = 1
		check2 = 0
		for i in range(1, len(s)):
			while i-check>=0 and check+i<=len(s):
				cur=s[i-check:check+i+1]
				if cur == cur[::-1]:
					res1 = cur
					check+=1
				else:
					break
			while i-1-check2>=0 and i+1+check2<=len(s):
				cur2=s[i-1-check2:i+1+check2]
				if cur2 == cur2[::-1]
					res2 = cur2
					check2 += 1
				else:
					break
		return res1 if len(res1)>len(res2) else res2
